CREATE DATABASE SUPERMERCADO

USE SUPERMERCADO

CREATE TABLE DISTRITOS(
CODDIS CHAR(4)PRIMARY ,
NOMDIS VARCHAR(25)NOT NUL
)
GO
CREATE TABLE CARGOS(
CODCAR CHAR(4)PRIMARY ,
NOCAR VARCHAR(30)NOT NUL)
GO
CREATE TABLE MARCAS(
CODMAR CHAR(4)PRIMARY 
NOMMAR VARCHAR(30)NOT NUL
)
GO
CREATE TABLE CATEGORIAS(
CODCAT CHAR(4) PRIMARY
NOMCAT VARCHAR(30)NOT NUL,
)
GO 
CREATE TABLE FORMA_PAGO(
CODFOR CHAR(4)PRIMARYKEY
NOMFOR VARCHAR(25)NOTNULL
)
GO
CREATE TABLE LOCALES(
CODLOC CHAR(4)PRIMARY KEY,
NOMLOC VARCHAR(30)NOT NUL,
DIREC VARCHAR(40)NOT NUL,
CODDIS CHAR(4)FOREIGN KEY REFERENSES DISTRITOS(CODDIS),
TELEF VARCHAR(9)NOT NUL,
EMAIL VARCHAR(30)NOT NUL
)








GO
CREATE TABLE CLIENTES(
CODCLI CHAR(4)PRIMARY ,
NOMCLI VARCHAR(40)NOT NUL,
DIREC VARCHAR(40)NOT NUL,
CODDIS CHAR(4)FOREIGN KEY REFERENSES DISTRITOS(CODDIS),
FNAC DATE NOT NUL,
SEXO CHAR(1)NOT NUL,
TELEF VARCHAR(9)NOT NUL
)
GO
CREATE TABLE PERSONAL(
CODPER CHAR(4)PRIMARY,
NOMPER VARCHAR(40)NOT NUL,
DIREC VARCHAR(40)NOT NUL,
CODDIS CHAR(4) FOREIGN KEY REFERENSES DISTRITOS(CODDIS),
FNAC  DATE NOT NUL,
SEXO CHAR(1)NOT NUL,
CODCAR CHAR(4)FOREIGN KEY REFERENSES CARGOS(CODCAR),
SUELDO DECIMAL(7.2)NOT NULL
)
GO
CREATE TABLE PRODUCTOS(
CODPROD CHAR(6)PRIMARY KEY,
NOMPROD VARCHAR(40)NOT NULL,
CODCAR CHAR(4)FOREIGN KEY REFERENCES MarCAS(CODMAR),
CODCAT CHAR(4)FOREIGN KEY REFERENCES CATEGORIAS(CODCAT),
PRECIO DECIMAL(7,2)NOT NUL
STOCK INT NOT NUL
)
GO
CREATE TABLE COMP_CABECERA(
CODCOM CHAR(6)PRIMARY ,
CODCLI CHAR(4)FOREIGN  REFERENSES CLIENTES(CODCLI),
CODPER CHAR(4)FOREIGN  REFERENSES PERSONAL(CODPER),
CODLOC CHAR(4)FOREIGN  REFERENSES LOCALES(CODLOC),
CODFOR CHAR(4)FOREIGN  REFERENSES FORMA_PAGO(CODFOR),
TIPO_COMP CHAR(1)NOT NULL,
FECHA DATE NOT NULL
)
GO
CREATE TABLE COMP_DETALLE(
CODCOM CHAR(6) FOREIGN REFERENSES COMP_CABECERA(CODCOM),
CODPROD CHAR(6)FOREIGN REFERENSES PRODUCTOS(CODPROD),
CANTIDAD INT NOT NUL,
PRECIO DECIMAL(7,2)NOT NUL,
DCTO DECIMAL(6,2)NOT NUL
)
